并查集基础

并查集

每个人有很多爱好，有相同爱好的在一个社交网络中，求有多少个社交网络，然后从大到小排序输出其中的人数

测试用例
8
3: 2 7 10
1: 4
2: 5 3
1: 4
1: 3
1: 4
4: 6 8 1 5
1: 4
输出结果
3
4 3 1    

#include <cstdio>

#include <iostream>

#include <vector>

#include <algorithm>

using namespace std;

int n;
int father[1005];
int num[1005];
vector<int > l[1005];
bool cmp(int a,int b)
{
    return a>b;
}
int findroot(int x)
{
    int a = x;
    while (x!=father[x])
        x=father[x];
    while (a!=father[a])
    {
        int z = a;
        a = father[a];
        father[z] = x;
    }
    return x;
    
}
int main()
{
    //freopen("./2.txt","r",stdin);
    scanf("%d",&n);
    for(int i = 0;i<1005;i++)
        father[i] = i;
    for(int i = 1;i<=n;i++)
    {
        int temp;
        int t2;

        scanf("%d", &temp);

        getchar();
        while (temp--)
        {
            scanf("%d", &t2);
            l[i].push_back(t2);
        }
    }
    for(int i =1;i<=n;i++)
    {
        int len = l[i].size();
        int r1 = findroot(l[i][0]);
        for (int j = 1; j < len; j++)
        {

            int r2 = findroot(l[i][j]);
            if (r1 != r2)
                father[r2] = r1;
        }
    }
    for(int i = 1;i<=n;i++)
    {
        num[findroot(l[i][0])]++;
    }
        

    sort(num,num+1005,cmp);
    int number = 0;
    for(int i = 0;i<1005;i++)
    {
        if(num[i]==0){number = i;break;}
    }
    printf("%d\n",number);
    for(int i = 0;i<number-1;i++)
        printf("%d ",num[i]);
    printf("%d\n",num[number-1]);
    return 0;
}
