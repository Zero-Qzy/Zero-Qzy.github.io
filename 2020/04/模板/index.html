void compare(int head,int mid,int tail)
{
    int temp[105];
    int i = head,j=mid+1;
    int k = head;
    while (i<=mid&&j<=tail)
    {
        if(a[i]<a[j])
            temp[k++] = a[i++];
        else
            temp[k++] = a[j++];
    }
    while (i<=mid)
        temp[k++] = a[i++];
    while (j<=tail)
        temp[k++] = a[j++];
    for(int i =head;i<=tail;i++)
        a[i] = temp[i];
}
void mergeSearch(int head,int tail)
{
    if(head<tail)
    {
        int mid=(head+tail)/2;
        mergeSearch(head,mid);
        mergeSearch(mid+1,tail);
        compare(head,mid,tail);
    }
}

void dfs(int root,int depth)
{
    int len = l[root].size();   
    if(len!=0)
    	for(int i =0;i<len;i++)
            dfs(l[root][i],depth+1);
}
void dfstrave(int root)
{
    int len = l[root].size();
    for(int i =0;i<len;i++)
        dfs(l[root][i],1);
}

图的最短路(dijkstra)

邻接矩阵

int g[505][505];//图
int roadw[505];//最短路
int mark[505];
int n;//节点个数
const int inf =0x3f3f3f3f;
void dijkstra(int root)
{
    for(int i = 0;i<n;i++)
        roadw[i] = inf;
    roadw[root] = 0;

    int minn = inf;
    int minnum = -1;
    
    for(int i = 0;i<n;i++)
    {
        minn = inf;
        minnum = -1;
        for(int j = 0;j<n;j++)
            if(mark[j]==0&&roadw[j]<minn)
            {
                minn = roadw[j];
                minnum = j;
            }
        if(minnum==-1)break;
        mark[minnum] = 1;
        for(int j =0;j<n;j++)
            if(mark[j]==0&&g[j][minnum]<inf&&g[j][minnum]+roadw[minnum]<roadw[j])
            {
                roadw[j]=g[j][minnum]+roadw[minnum];
            }
    }
}

int findroot(int x)
{
    int a = x;
    while (x!=father[x])
        x=father[x];
    while (a!=father[a])
    {
        int z = a;
        a = father[a];
        father[z] = x;
    }

}
